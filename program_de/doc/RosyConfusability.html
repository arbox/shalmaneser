 76:   def compute(splitID,     # string: split ID, may be nil
 77:               additionals) # array:string: "target", "target_pos", "gframe", "fgframe"
 78:     ###
 79:     # open and initialize stuff:
 80:     
 81:     # open database
 82:     database = Mysql.real_connect(@exp.get('host'), @exp.get('user'), 
 83:                                   @exp.get('passwd'), @exp.get('dbname'))
 84:     # make an object that creates views.
 85:     # read one frame at a time.
 86:     iterator = RosyIterator.new(database, @exp, "train",
 87:                                 "splitID" => splitID, 
 88:                                 "xwise" => "frame")
 89:     # get value for "no val"
 90:     noval = @exp.get("noval")
 91: 
 92:     counts_frame = Hash.new(0)
 93:     
 94:     # iterate through all frames and compute confusability of each FE
 95:     iterator.each_group { |group_descr_hash, frame|
 96:       
 97:       $stderr.puts "Computing confusability for #{frame}"
 98: 
 99:       # read all instances of the frame, columns: FE and GF
100:       view = iterator.get_a_view_for_current_group(["sentid","gold", "fn_gf",
101:                                                     "target","target_pos", "frame"])
102:       
103:       if additionals.include? "tmfframe"
104:         # find most frequent gframe for each target
105:         tmfframe = determine_target_most_frequent_sc(view, noval)
106:       end
107: 
108:       # count occurrences
109:       counts_gf = Hash.new(0)
110:       counts_fe = Hash.new(0)
111:       counts_gffe = Hash.new(0)
112: 
113:       view.each_sentence { |sentence|
114: 
115:         # make string consisting of all FN GFs of this sentence
116:         allgfs = Array.new()
117:         sentence.each { |inst|
118:           if inst["fn_gf"] != noval
119:             allgfs << inst["fn_gf"]
120:           end
121:         }
122: 
123:         # assume uniqueness of GFs
124:         # design decision, could also be done differently.
125:         # rationale: if a GF occurs more than once,
126:         # it's probable that this is because we get more than
127:         # one constituent for this GF, not because
128:         # it actually occurred more than once in the
129:         # original FrameNet annotation.
130:         allgfs.uniq!
131: 
132:         # now count each instance
133:         sentence.each { |row|
134:           if row["gold"] == "target"
135:             # don't count target among the FEs
136:             next
137:           end
138:         
139:           if row["gold"] != noval
140:             counts_fe[row["gold"]] += 1
141:           end
142:           if row["fn_gf"] != noval and row["fn_gf"] != "target"
143:             gf = row["fn_gf"]
144:             
145:             additionals.each { |additional|
146:               case additional
147:               when "target"
148:                 gf << "_" + row["target"]
149:               when "target_pos"
150:                 gf << "_" + row["target_pos"]
151:               when "gframe"
152:                 gf << "_" + allgfs.join("_")
153: 
154:               when "fgframe"
155:                 # find the maximal frequent frame subsuming allgfs
156:                 maxfgf = nil
157:                 @frequent_gframes.each { |fgframe|
158:                   if fgframe.subsumed_by?(allgfs)
159:                     # fgframe is a subset of allgfs
160:                     if maxfgf.nil? or fgframe.length() > maxfgf.length()
161:                       maxfgf = fgframe
162:                     end
163:                   end
164:                 }
165:                 if maxfgf.nil?
166:                   # nothing there that fits
167:                   # leave GF as is
168:                 else
169:                   gf << "_" + maxfgf.join("_")
170:                 end
171: 
172:               when "tmfframe"
173:                 gf << "_" + tmfframe[tmf_target_key(row)]
174: 
175:               else
176:                 raise "Don't know how to compute #{additional}"
177:               end
178:             }
179: 
180:             counts_gf[gf] += 1
181:           end
182:         
183:           if row["gold"] != noval and gf
184:             counts_gffe[gf + " " + row["gold"]] += 1
185:           end
186:         } # each row of sentence
187:       } # each sentence of view
188:       
189:       # compute gf entropy
190:       # gfe_{fr}(gf) = \sum_{fe \in fes(fr)} -p(fe|gf) log_2 p(fe|gf)
191:       # 
192:       # where p(fe|gf) = f(gf, fe) / f(gf)
193:       gf_entropy = Hash.new
194:       
195:       counts_gf.keys.each { |gf|
196:         gf_entropy[gf] = 0.0
197:         
198:         counts_fe.keys.each { |fe|
199:           if counts_gf[gf] > 0
200:             p_gf_fe = counts_gffe[gf + " " + fe].to_f / counts_gf[gf].to_f
201:             
202:             # get log_2 via log_10
203:             if p_gf_fe > 0.0
204:               gf_entropy[gf] -= p_gf_fe * Math.log10(p_gf_fe) * 3.32193
205:             end
206:           end
207:         } # each FE for this GF
208:       } # each GF (gf entropy)
209:       
210:       # compute FE confusability
211:       # c_{fr}(fe) = \sum_{gf \in gfs(fr)} p(gf|fe) gfe_{fr}(gf)
212:       #
213:       # where p(gf|fe) = f(gf, fe) / f(fe)
214:       counts_fe.keys.each { |fe|
215:         @confusability[frame + " " + fe] = 0.0
216:         
217:         counts_gf.keys.each { |gf|
218:           if counts_fe[fe] > 0
219:             p_fe_gf = counts_gffe[gf + " " + fe].to_f / counts_fe[fe].to_f
220:             
221:             @confusability[frame + " " + fe] += p_fe_gf * gf_entropy[gf]
222:           end
223:         } # each GF for this FE
224:       } # each FE (fe confusability)
225: 
226: 
227:       # remember counts for FEs and GF/FE pairs
228:       counts_fe.keys.each { |fe|
229:         @counts_fe_glob[frame + " " + fe] = counts_fe[fe]
230:       }
231:       counts_gffe.each_pair {|event,freq|
232:         @counts_gffe_glob[frame+" " +event] = freq
233:       }
234: 
235:       # omit rare FEs:
236:       # anything below 5 occurrences
237:       counts_fe.each_key {  |fe|
238:         if counts_fe[fe] < 5
239:           @confusability.delete(frame + " " + fe)
240:         end
241:       }
242: 
243:       # compute overall frame confusability
244:       # omitting rare FEs with below 5 occurrences:
245:       # 
246:       # c(fr) = sum_{fe \in fes(fr)} f(fe)/f(fr) * c_{fr}(fe)
247:       #       = \sum_{gf \in gfs(fr)} p(gf|fr) gfe_{fr}(gf)
248:       #
249:       # where p(gf|fr) = (sum_{fe\in fes(fr)} f(gf, fe)) / f(fr)
250:       counts_frame[frame] = 0
251:       counts_fe.each_value { |count|
252:         if count >= 5
253:           counts_frame[frame] += count
254:         end
255:       }
256:       @frame_confusability[frame] = 0.0
257:       counts_fe.each_pair { |fe, count|
258:         if count >= 5
259:           @frame_confusability[frame] += (count.to_f / counts_frame[frame].to_f) * @confusability[frame + " " + fe]
260:         end
261:       }
262:     } # each frame
263:     
264:     # compute overall confusability
265:     # c = \sum{fr \in frames} f(fr)/N * c(fr)
266:     #
267:     # where N is the number of FE occurrences overall
268:     counts_overall = 0
269:     counts_frame.each_value { |count|
270:       counts_overall += count
271:     }
272:     @overall_confusability = 0.0
273:     counts_frame.each_pair { |frame, count|
274:       @overall_confusability += (count.to_f / counts_overall.to_f) * @frame_confusability[frame]
275:     }
276:   end

compute sparseness statistics over the set of base events used for computing the confusability returns an array of length 4:

  • number of events with freq 1

  • number of events with freq 2

  • number of events with freq 3-5

  • number of events with freq > 5
