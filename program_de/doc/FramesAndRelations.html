133:   def add_relations(file)
134: 
135:     $stderr.puts "Reading frame and FE hierarchy"
136: 
137:     doc = Document.new(file)
138: 
139:     doc.root.elements.each("frame-relation-type") { |relation|
140: 
141:       case relation.attributes["name"]
142: 
143:       when "Inheritance", "Subframe", "Using", "See_also", "Inchoative_of", "Causative_of"
144: 
145:         $stderr.puts "Reading frame relation " + relation.attributes["name"]
146:         
147:         relation.elements.each("frame-relations/frame-relation") { |frame_rel|
148:           
149:           # enter relation info as graph edge from superframe to subframe
150:           superframe = read_oblig_attribute(frame_rel, "superFrameName")
151:           subframe = read_oblig_attribute(frame_rel, "subFrameName")
152:           
153:           add_frame_relation(relation.attributes["name"], superframe, subframe)
154:           
155:           # enter same relation for frame elements
156:           frame_rel.elements.each("fe-relation") { |fe_rel|
157:             
158:             superfe = read_oblig_attribute(fe_rel,"superFEName")
159:             subfe = read_oblig_attribute(fe_rel, "subFEName")
160:             
161:             ensure_fe_of_frame(superframe, superfe)
162:             ensure_fe_of_frame(subframe, subfe)
163:             
164:             add_fe_relation(relation.attributes["name"], superframe, superfe, subframe, subfe)
165:           }
166:         }
167: 
168:       when "Excludes", "Requires"
169:         $stderr.puts "Reading frame element relation " + relation.attributes["name"]
170:         
171:         relation.elements.each("frame-relations/frame-relation") { |frame_rel|
172:           # just one frame mentioned in frame_rel
173:           frame_name = read_oblig_attribute(frame_rel, "superFrameName")
174:           # this is really about the frame elements
175:           frame_rel.elements.each("fe-relation") { |fe_rel|
176:             
177:             superfe = read_oblig_attribute(fe_rel,"superFEName")
178:             subfe = read_oblig_attribute(fe_rel, "subFEName")
179:             
180:             ensure_fe_of_frame(frame_name, superfe)
181:             ensure_fe_of_frame(frame_name, subfe)
182:             
183:             add_fe_relation(relation.attributes["name"], frame_name, superfe, frame_name, subfe)
184:           }
185:         }

     # File PkgGeneral/FnHierarchy.rb, line 196
196:   def recover_from_dump()
197: 
198:     # construct method for recovering
199:     # a node by its ID
200:     proc_get_node = proc { |node_id|
201:       if @frames.has_key? node_id
202:         @frames[node_id]
203:       elsif @fes.has_key? node_id
204:         @fes[node_id]
205:       else
206:         $stderr.puts "Error: In trying to recover from dump,"
207:         $stderr.puts "encountered unknown node ID "+node_id.to_s
208:         exit 1
209:       end
210:     }
211: 
212:     # after pickling and restoring, all graph edges
213:     # of GraphNode objects point to node IDs rather
214:     # than to nodes. Restore the nodes from their IDs.
215:     [@frames, @fes].each { |hash|
216:       hash.each_value { |node|
217:         node.recover_from_dump(proc_get_node)
218:       }
219:     }
220:     
221:   end
